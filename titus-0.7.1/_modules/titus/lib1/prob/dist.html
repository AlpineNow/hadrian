    name = prefix + "gaussianLL"
    sig = Sigs([Sig([{"x": P.Double()}, {"mu": P.Double()}, {"sigma": P.Double()}], P.Double()),
                Sig([{"x": P.Double()}, {"params": P.WildRecord("A", {"mean": P.Double(), "variance": P.Double()})}], P.Double())])
    def __call__(self, state, scope, paramTypes, x, *others):
        if len(others) == 2:
            mu, sigma = others
        else:
            mu = others[0]["mean"]
            if others[0]["variance"] < 0.0:
                raise PFARuntimeException("invalid parameterization")
            else:
                sigma = math.sqrt(others[0]["variance"])
        return GaussianDistribution(mu, sigma).LL(x)

[docs]class GaussianDistribution(object):
    def __init__(self, mu, sigma):
        if sigma < 0.0:
            raise PFARuntimeException("invalid parameterization")
        self.mu = mu
        self.sigma = sigma


[docs]    def LL(self, x):
        if (self.sigma == 0.0) and (x == self.mu):
            return float("inf")
        elif (self.sigma == 0.0) and (x != self.mu):
            return float("-inf")
        else:
            term1 = -(x - self.mu)**2/(2.0 * self.sigma**2)
            term2 = math.log(self.sigma * math.sqrt(2.0 * math.pi))
            return term1 - term2



[docs]    def CDF(self, x):
        if (self.sigma == 0.0) and (x < self.mu):
            return 0.0
        elif (self.sigma == 0.0) and (x >= self.mu):
            return 1.0
        else:
            return 0.5 * (1.0 + math.erf((x - self.mu)/(self.sigma * math.sqrt(2.0))))



[docs]    def QF(self, p):
        if (p > 1.0) or (p < 0.0):
            raise PFARuntimeException("invalid input")
        elif (p == 1.0):
            return float("inf")
        elif (p == 0.0):
            return float("-inf")
        elif (self.sigma == 0.0):
            return self.mu
        else:
            # http://www.johnkerl.org/python/sp_funcs_m.py.txt
            c0 = 2.515517
            c1 = 0.802853
            c2 = 0.010328
            d1 = 1.432788
            d2 = 0.189269
            d3 = 0.001308

            sign = -1.0
            if (p > 0.5):
                sign = 1.0
                p = 1.0 - p

            arg = -2.0*math.log(p)
            t = math.sqrt(arg)
            g = t - (c0 + t*(c1 + t*c2)) / (1.0 + t*(d1 + t*(d2 + t*d3)))
            standard_normal_qf = sign*g
            return self.mu + self.sigma*standard_normal_qf

[docs]class ExponentialDistribution(object):
    def __init__(self, rate):
        self.rate = rate
        if (self.rate < 0.0):
            raise PFARuntimeException("invalid parameterization")


[docs]    def PDF(self, x):
        if (self.rate == 0.0):
            return 0.0
        elif (x < 0.0):
            return 0.0
        elif (x == 0.0):
            return self.rate
        else:
            return self.rate*math.exp(-self.rate*x)



[docs]    def CDF(self, x):
        if (self.rate == 0.0) or (x == 0.0):
            return 0.0
        elif (x <= 0.0):
            return 0.0
        else:
            return 1.0 - math.exp(-self.rate*x)



[docs]    def QF(self, p):
        if (p > 1.0) or (p < 0.0):
            raise PFARuntimeException("invalid input")
        elif (self.rate == 0.0) and (p == 0.0):
            return 0.0
        elif (self.rate == 0.0) and (p > 0):
            return float("inf")
        elif (p == 1.0):
            return float("inf")
        elif (p == 0.0):
            return 0.0
        else:
            return -math.log(1.0 - p)/self.rate

        elif (x == 0.0) and (self.DOF < 2):
            return float("inf")
        elif (x < 0.0):
            return 0.0
        else:
            return GammaDistribution(self.DOF/2.0, 2.0).PDF(x)

            return 0.0
        elif (x <= 0.0):
            return 0.0
        else:

[docs]    def PDF(self, x):
        if (self.lamda == 0.0):
            if (x != 0.0):
                return 0.0
            else:
                return 1.0
        elif (x < 0.0):
            return 0.0
        else:
            return math.exp(x * math.log(self.lamda) - self.lamda - math.lgamma(x + 1.0))



[docs]    def CDF(self, x):
        if math.isnan(x):
            return float("nan")
        elif (self.lamda == 0.0):
            if (x >= 0.0):
                return 1.0
            else:
